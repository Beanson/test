<script>

    var app = angular.module('myApp', []);
    app.controller('myCtrl', function ($scope, $http,DataSer) {

        $scope.supportData=DataSer.supportData;


    });

    app.factory('DataSer',function () {
        var supportData={
            'start_text_index':'',
            'end_text_index':'',
            'start_text_offset':'',
            'end_text_offset':''
        };

        return{
            supportData:supportData,
        }
    });


    app.directive('testSelect','DataSer',['$parse',function ($parse,DataSer) {
        return{
            restrict : 'A',
            scope:{

            },
            link: function(scope, element, attrs) {

                element.bind('mouseup', function(evt){
                    scope.$apply(function(){
                        var selectedRange = window.getSelection().getRangeAt(0);
                        DataSer.supportData['start_text_offset']=selectedRange.startOffset;
                        DataSer.supportData['end_text_offset']=selectedRange.startOffset;
                    });
                });
            }
        }
    }]);

    app.directive('testSelectSub','DataSer',['$parse',function ($parse,DataSer) {
        return{
            restrict : 'A',
            scope:{
                textId:'='
            },
            link: function(scope, element, attrs) {

                element.bind('mousedown', function(evt){
                    scope.$apply(function(){
                        DataSer.supportData['start_text_index']=scope.textId;
                    });
                });
                element.bind('mouseup', function(evt){
                    scope.$apply(function(){
                        DataSer.supportData['end_text_index']=scope.textId
                    });
                });
            }
        }
    }])

</script>
